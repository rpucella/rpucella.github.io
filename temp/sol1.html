(* Question 1 *)

let rec gcd (a:int) (b:int):int  = 
  if a < 0 || b < 0 then failwith "gcd: incorrect arguments"
  else if a = 0 then b
  else if b = 0 then a
  else if a > b then gcd b (a mod b)
  else gcd a (b mod a)
  
let rec selfConcat (s:string) (n:int):string =
  if n <= 0 then ""
  else s ^ (selfConcat s (n - 1))

let rec collatz (a:int):int list =
  if a = 1 then [1]
  else if a mod 2 = 0 then a :: (collatz (a / 2))
  else a :: (collatz (3 * a + 1))

let rec range (i:int) (j:int):int list =
  if i >= j then []
  else i :: (range (i + 1) j)


(* Question 2 *)

let rec squares (xs:int list):int list =
  match xs with
  | [] -> []
  | x::xs' -> (x * x) :: squares xs'

let rec prependString (x:string) (ys:string list):string list = 
   match ys with
   | [] -> []
   | y::ys -> (x ^ y) :: prependString x ys

let rec doubleUp (xs:int list):int list =
  match xs with
  | [] -> []
  | x::xs' -> x :: (2 * x) :: doubleUp xs'

let rec nonNegative (xs:int list):int list =
  match xs with
  | [] -> []
  | x::xs' -> if x >= 0 then x :: nonNegative xs' else nonNegative xs'

let rec concatAll (xs: 'a list list): 'a list =
  match xs with
  | [] -> []
  | x::xs' -> x @ concatAll xs'

let rec classify (xs:int list):int list * int list =
  match xs with
  | [] -> ([], [])
  | x::xs' -> (match classify xs' with
               | (p, n) -> if x >= 0 then (x :: p, n)
                           else (p, x :: n))

let rec scale (a:int) (v:int list):int list =
  match v with
  | [] -> []
  | x::xs -> (a * x) :: (scale a xs)


let rec add (v:int list) (w:int list):int list =
  match v with
  | [] -> []
  | x::xs -> (match w with
              | [] -> []
              | y::ys -> (x + y) :: (add xs ys))


let rec length (v:int list): float =
  let rec sum_squares xs =
    match xs with
    | [] -> 0
    | x::xs' -> (x * x) + sum_squares xs'  in
  sqrt (float (sum_squares v))

  
let rec inner (v:int list) (w:int list):int =
  match v with
  | [] -> 0
  | x::xs -> (match w with
              | [] -> 0
              | y::ys -> (x * y) + (inner xs ys))

let is_in = List.mem
          
let rec set (xs: 'a list): 'a list =
  match xs with
  | [] -> []
  | x::xs' -> if is_in x xs' then set xs' else x :: (set xs')

let rec set_sub (xs:'a list) (ys:'a list): bool = 
  match xs with 
  | [] -> true
  | x::xs' -> is_in x ys && set_sub xs' ys

let rec set_union (xs: 'a list) (ys: 'a list): 'a list = 
  match xs with
  | [] -> set ys
  | x::xs' -> let result = set_union xs' ys in
              if is_in x result then result
              else x :: result

let rec set_inter (xs:'a list) (ys:'a list): 'a list = 
  match xs with
  | [] -> []
  | x::xs' -> if (is_in x ys) && (not (is_in x xs'))
                then x :: (set_inter xs' ys)
              else set_inter xs' ys
